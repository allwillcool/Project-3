#include <stdio.h>
#include <windows.h>

// Функція для підрахунку кількості чисел із заданими цифрами
int countNumbers(int n) {
    if (n == 1) {
        return 2; // Для 1 цифри можливі два числа: 5 і 9
    }

    int count[2][n]; // Матриця для зберігання кількості чисел із цифрами
    count[0][0] = count[1][0] = 1; // Перша цифра може бути 5 або 9
    count[0][1] = count[1][1] = 2; // Друга цифра може бути 5 або 9, оскільки три послідовні однакові цифри не допускаються

    for (int i = 2; i < n; i++) {
        count[0][i] = count[0][i - 1] + count[1][i - 1]; // Кількість чисел з i цифрами, що закінчуються на 5
        count[1][i] = count[0][i - 1] + count[1][i - 1]; // Підрахунок чисел з i цифрами, що закінчуються на 9
    }

    return count[0][n - 1] + count[1][n - 1]; // Повертає загальну кількість чисел із n цифр
}

int main() {
SetConsoleCP(1251);
  SetConsoleOutputCP(1251);
    int n;
    printf("Введіть кількість цифр: ");
    scanf("%d", &n);

    int result = countNumbers(n);
    printf("Кількість можливих чисел: %d\n", result);

    return 0;
}
